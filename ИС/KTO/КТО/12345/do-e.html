#include <stdio.h>  
#include <math.h>  
  
/* Решение уравнения x-cos(sin(x))=0 */  
void main (void)  
{  
   double x_new=0.9, x_old, eps=0.0001;  
   do  
   {  
      x_old = x_new;  
      x_new = cos(sin(x_old));  
   }  while ( fabs( x_new - x_old ) > eps );  
   printf ( "x=%lf", x_new );  
}
