Оператор while является оператором цикла с предусловием.

Пример программы вычисления количества цифр в целом числе:

#include <iostream.h>
#include <conio.h>
#include <math.h>
void main()
{
int x; //число
int y; //модуль числа
int k=1; //количество цифр в числе
cout<<"x? "; cin>>x;
y=abs(x);
while (y>=10)
{
   k++;
   y/=10;  //целочисленное деление
}
cout<<"e;k="e;<<k;
getch();
}
