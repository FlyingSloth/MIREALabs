#include <iostream.h>

void main(void)

{ 
   int count; 
   int total = 0; 
   for (count = 1; count <= 100; count++) 
      { 
         cout << "Прибавляю " << count << " к " << total; 
         total = total + count; 
         cout << " получаю " << total << endl; 
       } 
}
